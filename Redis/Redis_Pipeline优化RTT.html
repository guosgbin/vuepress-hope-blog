                       port=6322,
                       db=0,
                       password='xxxx',
                       decode_responses=True)

    r = get_redis_cli()
    for i in range(100):
        r.ping()

def with_pipelining():
    r = get_redis_cli()
    with r.pipeline() as pipe:
        for i in range(100):
            pipe.ping()
        pipe.execute()
