    if (phaser.getPhase() != phase) {
        thread = null;
        return true;
    }
    // å½“å‰çº¿ç¨‹è¢«ä¸­æ–­äº†ï¼Œèµ‹å€¼ä¸­æ–­æ ‡è®°
    if (Thread.interrupted())
        wasInterrupted = true;
    // å‡å¦‚å½“å‰çº¿ç¨‹è¢«ä¸­æ–­äº†ï¼Œè€Œä¸”å½“å‰çº¿ç¨‹å°è£…çš„èŠ‚ç‚¹å…è®¸ä¸­æ–­ï¼Œé‚£ä¹ˆå°±ä¼šè¿”å› true
    if (wasInterrupted && interruptible) {
        thread = null;
        return true;
    }
    // æ”¯æŒè¶…æ—¶çš„åˆ¤æ–­
    if (timed) {
        if (nanos > 0L) {
            nanos = deadline - System.nanoTime();
        }
        if (nanos <= 0L) {
            thread = null;
            return true;
        }
    }
    return false;
}

public boolean block() {
    if (isReleasable())
        return true;
    else if (!timed)
        LockSupport.park(this);
    else if (nanos > 0L)
        LockSupport.parkNanos(this, nanos);
    return isReleasable();
}
